高级视图语法—循环

        //Foreach循环
        @foreach(var c in b)
        { 
            @c.Current.ToString()
        }

        //For循环
        @for(int  i = 0;i < b.Length;i++)
        {
            @b[i].ToString()
        }

        //While do循环
        @{int k = 0;}
        @while(k < b.Length;)
        {
            @b[k].ToString()
            @k++
        }
