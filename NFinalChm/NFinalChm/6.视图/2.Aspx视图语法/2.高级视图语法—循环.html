高级视图语法—循环

        <%//Foreach循环 %>
        <foreach enumerator="<%var c = b.GetEnumerator();%>">
            <%=c.Current.ToString() %>
        </foreach>

        <%//For循环 %>
        <for start="<%int  i = 0; %>" condition="<%= i < b.Length %>" step="<%i++;%>">
            <%=b[i].ToString() %>
        </for>

        <%//While do循环 %>
        <%int k = 0; %>
        <while condition="<%=k < b.Length; %>">
            <%=b[k].ToString(); %>
            <%k++; %>
        </while>
