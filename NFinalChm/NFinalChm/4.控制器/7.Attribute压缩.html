7.Attribute压缩

Gzip压缩 

        //urlPrefix="/App";
        namespace WebSite.App.Controllers
        {
            public class NewsController
            {
				//调用带有Zip后缀的Attribute路由，即实现Gzip压缩功能。
				[GetHtmlZip("~/Index.html")]
                public void Index()
                {
                    Write("Hello world!");
                }
            }
        }

Deflate压缩 

        //urlPrefix="/App";
        namespace WebSite.App.Controllers
