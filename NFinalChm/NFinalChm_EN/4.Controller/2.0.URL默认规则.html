路由配置

路由前缀，后缀
在模块目录下的config.json文件中。 

        //URL前缀
        "urlPrefix": "/App",
        //URL后缀
        "urlExtension": ".html",

默认路由规则

不带参数
URL=URL前缀（config.json中的urlPrefix项）
+"/"+类名+"/"+方法名
+URL后缀(config.json中的urlExtension项) 
        //如果此类的模块名为App
        namespace WebSite.App.Controllers
        {
            public class IndexController
            {
                public void Index()
                {
                    Write("Hello!");
                }
            }
        }
		//URL前缀为：/App
		//类名为:IndexController
		//方法名:Index
		//URL后缀为:.html
        //对应的URL为:http://localhost/App/IndexController/Index.html
带参数
URL=URL前缀（config.json中的urlPrefix项）
+"/"+类名+"/"+方法名
+"/"+参数1+"/"+参数2+"/"+参数3....
+两位数字(两位数字代表urlPrefix+类名+方法名的长度)
+URL后缀(config.json中的urlExtension项) 
        //如果此类的模块名为App
        namespace WebSite.App.Controllers
        {
            public class IndexController
            {
                public void Index(int id)
                {
                    Write("Hello!");
                }
            }
        }
		//URL前缀为：/App
		//类名为:User
		//方法名:Index
		//数字:15
		//参数1为：3
		//URL后缀为:.html
        //对应的URL为:http://localhost/App/User/Index/315.html
后台生成 
        //Url.cs类位于网站根目录下。
        string url=Url.IndexController_Index(int id);
        //模板调用时
        //<%=Urls.IndexController_Index(3);%>
前端生成 
        //需要引入网站根目录下的Url.js文件
        var id="1";
        var url=Url.IndexController_Index(int id);
