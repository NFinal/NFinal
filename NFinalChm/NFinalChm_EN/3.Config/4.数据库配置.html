数据库配置

1.配置Web.Config中的ConnectionString

mysql数据库: 
    "connectionStrings": [
		{
        	"name": "MySql",
		    "connectionString": "Server=localhost;Database=test;Uid=root;Pwd=root;",
		    "providerName": "MySql.Data.MySqlClient"
		}]
sqlite数据库: 
    "connectionStrings": [
		{
        	"name": "Sqlite",
		    "connectionString": "Data Source=|ModuleDataDirectory|Common.db;Pooling=true;FailIfMissing=false",
		    "providerName": "System.Data.SQLite"
		}]
sqlserver数据库: 
    "connectionStrings": [
		{
            "name": "SqlServer",
            "connectionString": "Server=.\\SQLEXPRESS;Database=test;Uid=sa;Pwd=123456;",
            "providerName": "System.Data.SqlClient"
		}]
oracle数据库: 
    "connectionStrings": [
		{
            "name": "Oracle",
            "connectionString": "Data Source==(DESCRIPTION=(ADDRESS_LIST=(ADDRESS=(PROTOCOL=TCP)(HOST=127.0.0.1)(PORT=1521)))(CONNECT_DATA=(SERVICE_NAME=ORCL)));User Id=c##sa;Password=123456;",
            "providerName": "Oracle.ManagedDataAccess.Client"
		}]
postgresql数据库: 
    "connectionStrings": [
		{
            "name": "PostgreSql",
            "connectionString": "Server=localhost;DataBase=postgres;Uid=postgres;Pwd=123456;",
            "providerName": "Npgsql"
		}]

2.运行WebCompile.aspx生成数据库操作类.

3.打开Models目录,显示所有文件,把生成的数据库操作类添加进项目中.
